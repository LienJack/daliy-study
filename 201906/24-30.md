> 20. 有效的括号
> 给定一个只包括 '('，')'，'{'，'}'，'['，']' 的字符串，判断字符串是否有效。

> 有效字符串需满足：左括号必须用相同类型的右括号闭合。左括号必须以正确的顺序闭合。注意空字符串可被认为是有效字符串。

```javascript
var isValid = function(s) {
  let stach = []
  let map = { '{':'}', "[": "]","(":")" }  
  let arr = s.split("")
  for(let i = 0; i < arr.length; i++) {
    if(arr[i] in map) {
      stach.push(arr[i])
    } else {
      let tmp = stach.pop()
      if (map[tmp]!==arr[i]) return false 
    }
  } 
  if (!stach.length) {
    return true
  } else {
    return false
  }
    
};

```
